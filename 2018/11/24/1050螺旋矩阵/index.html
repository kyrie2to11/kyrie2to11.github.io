1050螺旋矩阵

  1. 螺旋矩阵(25)
    时间限制150 ms
    内存限制
    65536 kB
    代码长度限制
    8000 B
    判题程序
    Standard
    作者
    CHEN, Yue
    本题要求将给定的N个正整数按非递增的顺序，填入“螺旋矩阵”。所谓“螺旋矩阵”，是指从左上角第1个格子开始，按顺时针螺旋方向填充。要求矩阵的规模为m行n列，满足条件：m*n等于N；m>=n；且m-n取所有可能值中的最小值。
    输入格式：
    输入在第1行中给出一个正整数N，第2行给出N个待填充的正整数。所有数字不超过104，相邻数字以空格分隔。
    输出格式：
    输出螺旋矩阵。每行n个数字，共m行。相邻数字以1个空格分隔，行末不得有多余空格。
    输入样例：
    12
    37 76 20 98 76 42 53 95 60 81 58 93
    输出样例：
    98 95 93
    42 37 81
    53 20 76
    58 60 76

解法一：难点有两个，一个算出m和n，另一个是螺旋填充。思路是：

用四个变量isEast、isSouth…来记录当前行径方向，通过判断前进方向是否满足条件来转变方向。要注意的是，在判断是否可以按当前前进方向继续前进的时候，要预先往前进一位，看看这个位是否超出或者已被填充，那么就要求进位之后不能超出数二维数组范围，于是建立一个(m+2)(n+2)的二维矩阵，填充的是中间mn的部分，这样就不会出现越界情况。

#include

#include

#include

#include
using namespace std;
bool cmp(int a, int b){return a > b;}
int main(){
int num;
cin>>num;
int m,n;
double d=sqrt((double)num);//强制类型转化需给类型符加上‘()’,sqrt()参数和返回值均为double。
if ((d-(int)d) != 0){
for (int i = 1; i <= (int)d; i++){
if ((num % i) != 0)
continue;
else {
n = i;
m = num / n;
}
}
} else m = n = (int)d;
vector arr(num, 0);
for (int i = 0; i < num; i++)
cin >> arr[i];
sort(arr.begin(), arr.end(), cmp);
vector< vector > show(m+2, vector(n+2, 0)); //多两位，防止访问非法
int i= 1, j = 1; //ii行，jj列
int isEast = 1, isSouth = 0, isWest = 0, isNorth = 0;
for (int k = 1; k <= num; k++){
show[i][j] = arr[k - 1];
if (isEast&&(j==n||show[i][j+1]!= 0)){
isEast = 0;
isSouth = 1;
} else if (isSouth&&(i==m||show[i+1][j] != 0)){
isSouth = 0;
isWest = 1;
} else if (isWest&&(j==1||show[i][j-1]!=0)){
isWest = 0;
isNorth = 1;
} else if (isNorth&&(i==1||show[i-1][j]!= 0)){
isNorth = 0;
isEast = 1;
}
if (isEast) j++;
else if (isSouth) i++;
else if (isWest) j–;
else if (isNorth) i–;
}
for (int i = 1; i <= m; i++){
for (int j = 1; j <= n; j++){
if (j != 1) printf(“ “);
cout << show[i][j];
}
cout << endl;
}
return 0;
}

解法二：螺旋填充路线对应矩阵下标关系依次填充。

#include

#include

#include

#include
using namespace std;
int func(int N) {//寻找N=m*n,m-n最小
int i = sqrt((double)N);
while (i >= 1) {
if (N % i == 0)
return i;
i–;
}
return 1;
}
int cmp(int a, int b) { return a > b; }//cmp 区分不了’=’
int main() {
int N, m, n, t = 0;
scanf(“%d”, &N);
n = func(N);
m = N / n;
vector a(N);
for (int i = 0; i < N; i++)
scanf(“%d”, &a[i]);
sort(a.begin(), a.end(), cmp);
vector<vector > b(m, vector(n));//vectorb[m][n];编译出错
int level = m / 2 + m % 2;
for (int i = 0; i < level; i++) {//加上t<=N-1的作用：应对奇数行时后三循环无效，因此第一个循环可以不加t<=N-1。
for (int j = i; j <= n - 1 - i&& t <= N - 1; j++) {
b[i][j] = a[t++];
}
for (int j = i + 1; j <= m - 2 - i && t <= N - 1; j++) {
b[j][n - 1 - i] = a[t++];
}
for (int j = n - i - 1; j >= i && t <= N - 1; j–) {
b[m - 1 - i][j] = a[t++];
}
for (int j = m - 2 - i; j >= i + 1 && t <= N - 1; j–) {
b[j][i] = a[t++];
}
}
for (int i = 0; i < m; i++) {
for (int j = 0; j < n; j++) {
if (j != 0) printf(“ “);
printf(“%d”, b[i][j]);
}
printf(“\n”);
}
return 0;
}
总结：解法二的求m,n的方法理论上用时更少更简单，只要找到一次n%i==0即可，而解法一可能需要多次n%i==0保留最后一次。
螺旋填充个人认为解法一的更有趣一点，解法一下标与螺旋关系设置循环初始值、结尾值次数较多易出错。
